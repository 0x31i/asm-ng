CONFIGURATION:
- CSV: Auto-detect (only .csv file in ./obfuscated/) -- ASM-NG export
- Format: Auto-detect 7-col (standard) or 10-col (full_scored with CVR scores)
- Nessus File: Auto-detect (only .nessus file in ./obfuscated/) - for EXTERNAL_VULNERABILITIES enrichment
- Mode: PRODUCTION (analyze ALL rows)
- Output: ./output/
- Data: TOKENIZED/OBFUSCATED (IPs, domains, hosts are token identifiers)

CONSOLIDATION RULES (MANDATORY):
The goal is a concise, executive-readable report. Minimize finding count through
aggressive consolidation while preserving detail in supporting CSVs.
- NEVER create one finding per individual port, IP, domain, or asset instance.
- Group related findings by EVENT TYPE + SEVERITY TIER (or risk tier for ports).
- Open ports MUST be consolidated into at most 3 findings (Critical-Risk, High-Risk,
  Standard) regardless of how many unique ports exist. Detail CSVs list every port/host.
- For all other event types, the consolidation phase groups by (Category, Vuln_Type,
  Priority) so multiple instances of the same event type at the same severity merge
  into a single finding with instance count and asset list.
- Target: under 20 consolidated findings for this group. If consolidation produces
  more than 20, review Vuln_Type keys for over-granularity.

def normalize_priority(severity, default='Medium'):
    """Normalize severity to standard format"""
    if not severity or str(severity).lower().strip() in ['nan', '', 'none']:
        return default
    severity = str(severity).strip().lower()
    if severity in ['critical', 'crit', '4']:
        return 'Critical'
    elif severity in ['high', 'hi', '3']:
        return 'High'
    elif severity in ['medium', 'med', '2']:
        return 'Medium'
    elif severity in ['low', 'lo', '1']:
        return 'Low'
    elif severity in ['info', 'informational', '0']:
        return 'Info'
    return default

def get_cvr_scores(row):
    """Extract CVR scores from a row (10-col format only)."""
    if CSV_FORMAT != 'full_scored':
        return None, None, None
    try:
        confidence = int(row.get('Confidence', 0) or 0)
        visibility = int(row.get('Visibility', 0) or 0)
        risk = int(row.get('Risk', 0) or 0)
        return confidence, visibility, risk
    except (ValueError, TypeError):
        return 0, 0, 0

def get_tracking(row):
    """Extract tracking status from row."""
    return str(row.get('Tracking', 'OPEN')).strip().upper() or 'OPEN'

def get_fp(row):
    """Extract FP value. Returns int: 0=unverified, 1=false positive, 2=verified."""
    try:
        return int(row.get('F/P', 0) or 0)
    except (ValueError, TypeError):
        return 0

def parse_severity_from_data(data_field, default='Medium'):
    """Parse severity level from the Data field text."""
    data_lower = str(data_field).lower()
    if 'critical' in data_lower:
        return 'Critical'
    elif 'high' in data_lower:
        return 'High'
    elif 'medium' in data_lower or 'med' in data_lower:
        return 'Medium'
    elif 'low' in data_lower:
        return 'Low'
    elif 'info' in data_lower or 'informational' in data_lower:
        return 'Info'
    return default

def load_nessus_data(nessus_file_path):
    """Parse .nessus XML file and build lookup dictionary for enrichment."""
    print(f"\n=== Loading Nessus Data ===")
    print(f"Parsing: {nessus_file_path}")

    nessus_data = {}
    plugin_info = {}

    try:
        tree = ET.parse(nessus_file_path)
        root = tree.getroot()
        hosts_processed = 0
        findings_indexed = 0

        for report_host in root.findall('.//ReportHost'):
            host_name = report_host.get('name', '')
            hosts_processed += 1
            hostname = None
            host_ip = host_name

            for tag in report_host.findall('.//HostProperties/tag'):
                tag_name = tag.get('name', '')
                if tag_name == 'hostname':
                    hostname = tag.text
                elif tag_name == 'host-ip':
                    host_ip = tag.text

            for item in report_host.findall('.//ReportItem'):
                plugin_id = item.get('pluginID', '')
                plugin_name = item.get('pluginName', '')
                port = item.get('port', '0')
                protocol = item.get('protocol', 'tcp')
                severity = item.get('severity', '0')
                description = item.findtext('description', '').strip()
                solution = item.findtext('solution', '').strip()
                risk_factor = item.findtext('risk_factor', '').strip()
                cvss_score = item.findtext('cvss3_base_score', '')
                if not cvss_score:
                    cvss_score = item.findtext('cvss_base_score', '')
                cves = [cve.text for cve in item.findall('cve') if cve.text]
                cve_string = ', '.join(cves) if cves else ''
                plugin_output = item.findtext('plugin_output', '').strip()
                see_also = item.findtext('see_also', '').strip()

                enrichment = {
                    'plugin_id': plugin_id, 'plugin_name': plugin_name,
                    'host': host_ip, 'hostname': hostname,
                    'port': port, 'protocol': protocol, 'severity': severity,
                    'description': description, 'solution': solution,
                    'risk_factor': risk_factor, 'cvss_score': cvss_score,
                    'cve': cve_string, 'plugin_output': plugin_output,
                    'see_also': see_also
                }

                if plugin_id and plugin_id not in plugin_info:
                    plugin_info[plugin_id] = {
                        'description': description, 'solution': solution,
                        'plugin_name': plugin_name, 'risk_factor': risk_factor
                    }

                nessus_data[f"{host_ip}:{port}:{plugin_id}"] = enrichment
                if hostname:
                    nessus_data[f"{hostname}:{port}:{plugin_id}"] = enrichment
                if f"{host_ip}:{plugin_id}" not in nessus_data:
                    nessus_data[f"{host_ip}:{plugin_id}"] = enrichment
                nessus_data[f"{host_ip}:{port}:{plugin_name.lower().replace(' ', '_')}"] = enrichment
                findings_indexed += 1

        print(f"Hosts processed: {hosts_processed}")
        print(f"Findings indexed: {findings_indexed}")
        print(f"Unique plugins: {len(plugin_info)}")
        print(f"Lookup keys created: {len(nessus_data)}")
        nessus_data['_plugin_info'] = plugin_info
        return nessus_data

    except Exception as e:
        print(f"ERROR: Failed to load Nessus file: {e}")
        return {}


def get_nessus_enrichment(nessus_data, host, port, plugin_id, plugin_name):
    """Look up Nessus enrichment data using multiple fallback strategies."""
    if not nessus_data:
        return None
    host = str(host).strip()
    port = str(port).strip() if port else '0'
    plugin_id = str(plugin_id).strip() if plugin_id else ''
    plugin_name = str(plugin_name).strip() if plugin_name else ''

    for key in [f"{host}:{port}:{plugin_id}", f"{host}:{plugin_id}",
                f"{host}:{port}:{plugin_name.lower().replace(' ', '_')}"]:
        if key in nessus_data:
            return nessus_data[key]

    if plugin_id and '_plugin_info' in nessus_data:
        if plugin_id in nessus_data['_plugin_info']:
            pi = nessus_data['_plugin_info'][plugin_id]
            return {'description': pi['description'], 'solution': pi['solution'],
                    'plugin_name': pi['plugin_name'], 'risk_factor': pi['risk_factor'],
                    'plugin_output': '', 'cve': '', 'cvss_score': ''}
    return None


def format_nessus_description(enrichment, host, port, plugin_name, severity):
    """Format Nessus description for output. Ensures 3-5 sentences."""
    if not enrichment or not enrichment.get('description'):
        return None
    raw = ' '.join(enrichment['description'].split())
    desc = f"A {severity.lower()} severity vulnerability '{plugin_name}' was identified on {host}"
    if port and port not in ['0', 'nan', '']:
        desc += f" port {port}"
    desc += ". "
    if len(raw) > 800:
        raw = raw[:800]
        lp = raw.rfind('.')
        raw = raw[:lp+1] if lp > 400 else raw + "..."
    desc += raw + " "
    cve = enrichment.get('cve', '')
    cvss = enrichment.get('cvss_score', '')
    if cve and cve not in ['nan', '']:
        desc += f"This vulnerability is tracked as {cve}. "
    if cvss and cvss not in ['nan', '']:
        desc += f"CVSS Score: {cvss}. "
    return desc


def format_nessus_recommendation(enrichment, host, severity):
    """Format Nessus solution/recommendation. Ensures 3-5 sentences."""
    if not enrichment or not enrichment.get('solution'):
        return None
    raw = ' '.join(enrichment['solution'].split())
    if len(raw) > 600:
        raw = raw[:600]
        lp = raw.rfind('.')
        raw = raw[:lp+1] if lp > 300 else raw + "..."
    rec = raw + " "
    rec += "Verify remediation by re-scanning with vulnerability assessment tools. "
    rec += f"Priority: {severity} - address according to organizational vulnerability management SLAs. "
    see_also = enrichment.get('see_also', '')
    if see_also and see_also not in ['nan', ''] and len(see_also) < 200:
        rec += f"Reference: {see_also.split()[0] if ' ' in see_also else see_also}"
    return rec

        source = str(row.get('Source', '')).strip()
        module = str(row.get('Module', '')).strip()

        confidence, visibility, risk = get_cvr_scores(row)
        if confidence is not None:
            group_stats['cvr_scores'].append((confidence, visibility, risk))

        port_match = re.search(r'(\d+)', data_field)
        port_num = port_match.group(1) if port_match else ''

        severity = default_sev
        svc = ''
        if port_num in HIGH_RISK_PORTS:
            svc, severity = HIGH_RISK_PORTS[port_num]

        port_tiers[severity].append({
            'port': port_num, 'svc': svc, 'host': source,
            'proto': proto, 'tracking': tracking,
            'risk': risk, 'data': data_field[:200], 'module': module,
        })

# Emit ONE consolidated finding per non-empty risk tier
for tier_sev in ['Critical', 'High', 'Medium']:
    entries = port_tiers[tier_sev]
    if not entries:
        continue

    unique_ports = sorted(set(e['port'] for e in entries if e['port']),
                          key=lambda x: int(x) if x.isdigit() else 0)
    unique_hosts = sorted(set(e['host'] for e in entries if e['host']))
    tracking_counts = {}
    for e in entries:
        t = e['tracking']
        tracking_counts[t] = tracking_counts.get(t, 0) + 1

    # Build port summary with service names
    port_details = []
    for p in unique_ports:
        if p in HIGH_RISK_PORTS:
            port_details.append(f"{p}/{HIGH_RISK_PORTS[p][0]}")
        else:
            port_details.append(p)

    if len(port_details) <= 20:
        port_list_str = ", ".join(port_details)
    else:
        port_list_str = ", ".join(port_details[:20]) + f", and {len(port_details) - 20} more"

    host_list_str = ", ".join(unique_hosts[:5])
    if len(unique_hosts) > 5:
        host_list_str += f", and {len(unique_hosts) - 5} more"

    tier_label = {'Critical': 'Critical-Risk', 'High': 'High-Risk', 'Medium': 'Standard'}[tier_sev]

    description = (f"{len(unique_ports)} {tier_label.lower()} open ports detected across "
                   f"{len(unique_hosts)} hosts ({len(entries)} total port/host instances). "
                   f"Ports: {port_list_str}. Hosts: {host_list_str}. ")
    if tier_sev == 'Critical':
        description += ("These include database, remote access, and administrative services "

        description += ("These include sensitive services that should not be directly accessible "

    else:
        description += ("Each exposed port increases attack surface. Review to ensure only "

        'Tab': 'TCP_PORT_OPEN',
        'Priority': tier_sev,
        'Item': f"{tier_label} Open Ports - {len(unique_ports)} ports across {len(unique_hosts)} hosts",

    group_stats['findings_extracted'] += len(entries)

                'Instance': i + 1, 'Port': e['port'], 'Protocol': e['proto'],
                'Service': HIGH_RISK_PORTS.get(e['port'], ('',))[0] if e['port'] in HIGH_RISK_PORTS else '',
                'Host': e['host'], 'Risk_Tier': tier_sev,
                'Tracking_Status': e['tracking'], 'Raw_Data': e['data'],
            })

for key, group_findings in groups.items():
    if len(group_findings) == 1:
        consolidated_findings.append(group_findings[0])
    else:
        category, vuln_type, priority = key
        assets = [f['Asset'] for f in group_findings]
        unique_assets = list(set([a for a in assets if a and a.lower() not in ['nan', 'none']]))
        asset_count = len(unique_assets)
        instance_count = len(group_findings)

        asset_list = ", ".join([f"({i+1}) {unique_assets[i]}" for i in range(min(5, len(unique_assets)))])
        if len(unique_assets) > 5:
            asset_list += f", and {len(unique_assets)-5} more"

        tracking_counts = defaultdict(int)
        for f in group_findings:
            tracking_counts[f.get('Tracking_Status', 'OPEN')] += 1
        tracking_summary = ", ".join([f"{k}: {v}" for k, v in tracking_counts.items()])

        avg_risk = None
        risk_scores = [f.get('Avg_Risk') for f in group_findings if f.get('Avg_Risk') is not None]
        if risk_scores:
            avg_risk = sum(risk_scores) / len(risk_scores)

        vuln_name = vuln_type.replace('_', ' ').title()
        _lead = f"{vuln_name} was identified on {instance_count} instances across {asset_count} assets: {asset_list}."

            f"Prioritize remediation based on asset criticality and verify each of the "
            f"{asset_count} affected assets through targeted testing."
        )

            'Category': category,
