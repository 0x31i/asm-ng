CONSOLIDATION RULES (MANDATORY):
The goal is a concise, executive-readable report. Minimize finding count through
aggressive consolidation while preserving detail in supporting CSVs.
- NEVER create one finding per individual port, IP, domain, URL, or asset instance.
- Group related findings by EVENT TYPE + SEVERITY TIER.
- The consolidation phase groups by (Category, Vuln_Type, Priority) so multiple
  instances of the same event type at the same severity merge into a single finding
  with instance count and asset list.
- Detail CSVs capture every individual instance for drill-down.
- Target: under 20 consolidated findings per group. If consolidation produces more
  than 20, review Vuln_Type keys for over-granularity.

# Auto-detect CSV format (7-col standard or 10-col full_scored)
def detect_csv_format(csv_file):
    with open(csv_file, 'r') as f:
        reader = csv.reader(f)
        headers = next(reader)
        headers_lower = [h.strip().lower() for h in headers]
        if 'confidence' in headers_lower:
            return 'full_scored', headers
        else:
            return 'standard', headers

CSV_FORMAT, CSV_HEADERS = detect_csv_format(CSV_FILE)

# Load and group by event type
def load_csv_by_event_type(csv_file):
    rows_by_type = {}
    with open(csv_file, 'r') as f:
        reader = csv.DictReader(f)
        for row in reader:
            etype = row['Type']
            rows_by_type.setdefault(etype, []).append(row)
    return rows_by_type

def normalize_priority(severity, default='Medium'):
    """Normalize severity to standard format"""
    if not severity or str(severity).lower().strip() in ['nan', '', 'none']:
        return default
    severity = str(severity).strip().lower()
    if severity in ['critical', 'crit', '4']:
        return 'Critical'
    elif severity in ['high', 'hi', '3']:
        return 'High'
    elif severity in ['medium', 'med', '2']:
        return 'Medium'
    elif severity in ['low', 'lo', '1']:
        return 'Low'
    elif severity in ['info', 'informational', '0']:
        return 'Info'
    return default

def get_cvr_scores(row):
    """Extract CVR scores from a row (10-col format only)."""
    if CSV_FORMAT != 'full_scored':
        return None, None, None
    try:
        confidence = int(row.get('Confidence', 0) or 0)
        visibility = int(row.get('Visibility', 0) or 0)
        risk = int(row.get('Risk', 0) or 0)
        return confidence, visibility, risk
    except (ValueError, TypeError):
        return 0, 0, 0

def get_tracking(row):
    """Extract tracking status from row."""
    return str(row.get('Tracking', 'OPEN')).strip().upper() or 'OPEN'

def get_fp(row):
    """Extract FP value. Returns int: 0=unverified, 1=false positive, 2=verified."""
    try:
        return int(row.get('F/P', 0) or 0)
    except (ValueError, TypeError):
        return 0

print("\n=== CONSOLIDATION PHASE ===")

groups = defaultdict(list)
for f in all_findings:
    key = (f['Category'], f['Vuln_Type'], f['Priority'])
    groups[key].append(f)

print(f"Consolidation groups: {len(groups)}")
for key, findings in groups.items():
    if len(findings) > 1:
        print(f"  {key}: {len(findings)} findings")

consolidated_findings = []
os.makedirs('./output/findings_details', exist_ok=True)

for key, group_findings in groups.items():
    if len(group_findings) == 1:
        consolidated_findings.append(group_findings[0])
    else:
        category, vuln_type, priority = key
        assets = [f['Asset'] for f in group_findings]
        unique_assets = list(set([a for a in assets if a and a.lower() not in ['nan', 'none']]))
        asset_count = len(unique_assets)
        instance_count = len(group_findings)

        asset_list = ", ".join([f"({i+1}) {unique_assets[i]}" for i in range(min(5, len(unique_assets)))])
        if len(unique_assets) > 5:
            asset_list += f", and {len(unique_assets)-5} more"

        tracking_counts = defaultdict(int)
        for f in group_findings:
            tracking_counts[f.get('Tracking_Status', 'OPEN')] += 1
        tracking_summary = ", ".join([f"{k}: {v}" for k, v in tracking_counts.items()])

        avg_risk = None
        risk_scores = [f.get('Avg_Risk') for f in group_findings if f.get('Avg_Risk') is not None]
        if risk_scores:
            avg_risk = sum(risk_scores) / len(risk_scores)

        vuln_name = vuln_type.replace('_', ' ').title()
        _lead = f"{vuln_name} was identified on {instance_count} instances across {asset_count} assets: {asset_list}."
        _default_desc = (
            _lead + " This represents a systematic security issue affecting multiple systems "
            "requiring coordinated remediation. The widespread nature indicates common "
            "misconfiguration, shared vulnerable components, or consistent security control gaps."
        )
        _default_rec = (
            f"Implement systematic remediation across all {instance_count} affected instances. "
            "Deploy consistent security controls and monitoring across all affected assets. "
            f"Prioritize remediation based on asset criticality and verify each of the "
            f"{asset_count} affected assets through targeted testing."
        )
        description, recommendation = _enrich(vuln_type, _lead, _default_desc, _default_rec)

        consolidated = {
            'Category': category,
            'Tab': group_findings[0]['Tab'],
            'Priority': priority,
            'Item': f"{vuln_name} - {instance_count} instances across {asset_count} assets",
            'Description': description,
            'Recommendation': recommendation,
            'Vuln_Type': vuln_type,
            'Asset': ", ".join(unique_assets[:5]) if unique_assets else '',
            'Plugin_Name': group_findings[0].get('Plugin_Name', ''),
            'Tracking_Status': max(tracking_counts, key=tracking_counts.get),
        }
        if avg_risk is not None:
            consolidated['Avg_Risk'] = round(avg_risk, 1)
        consolidated_findings.append(consolidated)

        # Create detail file
        import csv as csv_mod
        detail_data = [{'Instance': i+1, 'Asset': f['Asset'],
                        'Details': f['Item'], 'Description': f['Description'][:500],
                        'Tracking_Status': f.get('Tracking_Status', 'OPEN'),
                        'FP_Status': f.get('FP_Status', 'Unverified')}
                       for i, f in enumerate(group_findings)]
        detail_file = f'./output/findings_details/findings_detail_{vuln_type}.csv'
        with open(detail_file, 'w', newline='') as csvf:
            writer = csv_mod.DictWriter(csvf, fieldnames=detail_data[0].keys())
            writer.writeheader()
            writer.writerows(detail_data)
        print(f"  Created: {detail_file}")

print(f"\nConsolidation complete: {len(all_findings)} -> {len(consolidated_findings)} findings")
