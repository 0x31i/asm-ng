- Detail CSVs capture every individual instance for drill-down.
- Target: under 20 consolidated findings per group. If consolidation produces more

def normalize_priority(severity, default='Medium'):
    """Normalize severity to standard format"""
    if not severity or str(severity).lower().strip() in ['nan', '', 'none']:
        return default
    severity = str(severity).strip().lower()
    if severity in ['critical', 'crit', '4']:
        return 'Critical'
    elif severity in ['high', 'hi', '3']:
        return 'High'
    elif severity in ['medium', 'med', '2']:
        return 'Medium'
    elif severity in ['low', 'lo', '1']:
        return 'Low'
    elif severity in ['info', 'informational', '0']:
        return 'Info'
    return default

def get_cvr_scores(row):
    """Extract CVR scores from a row (10-col format only)."""
    if CSV_FORMAT != 'full_scored':
        return None, None, None
    try:
        confidence = int(row.get('Confidence', 0) or 0)
        visibility = int(row.get('Visibility', 0) or 0)
        risk = int(row.get('Risk', 0) or 0)
        return confidence, visibility, risk
    except (ValueError, TypeError):
        return 0, 0, 0

def get_tracking(row):
    """Extract tracking status from row."""
    return str(row.get('Tracking', 'OPEN')).strip().upper() or 'OPEN'

def get_fp(row):
    """Extract FP value. Returns int: 0=unverified, 1=false positive, 2=verified."""
    try:
        return int(row.get('F/P', 0) or 0)
    except (ValueError, TypeError):
        return 0

def parse_severity_from_data(data_field, default='Medium'):
    """Parse severity level from the Data field text."""
    data_lower = str(data_field).lower()
    if 'critical' in data_lower:
        return 'Critical'
    elif 'high' in data_lower:
        return 'High'
    elif 'medium' in data_lower or 'med' in data_lower:
        return 'Medium'
    elif 'low' in data_lower:
        return 'Low'
    elif 'info' in data_lower or 'informational' in data_lower:
        return 'Info'
    return default

class BurpHTMLParser(HTMLParser):
    """Parse Burp Suite HTML report to extract vulnerability details."""

    def __init__(self):
        super().__init__()
        self.vulnerabilities = []
        self.current_vuln = {}
        self.current_tag = ''
        self.current_class = ''
        self.capture_text = False
        self.captured_text = ''
        self.in_issue_detail = False
        self.in_issue_background = False
        self.in_remediation_detail = False
        self.in_remediation_background = False
        self.in_severity = False
        self.in_issue_name = False
        self.in_url = False
        self.section_depth = 0

    def handle_starttag(self, tag, attrs):
        attrs_dict = dict(attrs)
        self.current_tag = tag
        self.current_class = attrs_dict.get('class', '')

        # Detect Burp report sections by common class/id patterns
        if tag in ['h1', 'h2', 'h3'] and any(c in self.current_class.lower() for c in ['issue-name', 'vulnerability', 'finding']):
            self.in_issue_name = True
            self.captured_text = ''
        elif 'issue-detail' in self.current_class.lower() or 'vulnerability-detail' in self.current_class.lower():
            self.in_issue_detail = True
            self.captured_text = ''
        elif 'issue-background' in self.current_class.lower() or 'vulnerability-description' in self.current_class.lower():
            self.in_issue_background = True
            self.captured_text = ''
        elif 'remediation-detail' in self.current_class.lower() or 'remediation' in self.current_class.lower():
            self.in_remediation_detail = True
            self.captured_text = ''
        elif 'remediation-background' in self.current_class.lower():
            self.in_remediation_background = True
            self.captured_text = ''
        elif 'severity' in self.current_class.lower():
            self.in_severity = True
            self.captured_text = ''
        elif tag == 'a' and 'url' in self.current_class.lower():
            self.in_url = True
            self.captured_text = ''

        # Also detect table-based Burp reports
        if tag == 'span' and 'ISSUE_SEVERITY' in attrs_dict.get('class', '').upper():
            self.in_severity = True
            self.captured_text = ''

    def handle_endtag(self, tag):
        text = self.captured_text.strip()

        if self.in_issue_name and tag in ['h1', 'h2', 'h3']:
            if self.current_vuln:
                self.vulnerabilities.append(self.current_vuln)
            self.current_vuln = {'name': text, 'severity': '', 'detail': '',
                                 'background': '', 'remediation': '', 'url': ''}
            self.in_issue_name = False
        elif self.in_issue_detail and tag == 'div':
            if self.current_vuln:
                self.current_vuln['detail'] = text
            self.in_issue_detail = False
        elif self.in_issue_background and tag == 'div':
            if self.current_vuln:
                self.current_vuln['background'] = text
            self.in_issue_background = False
        elif self.in_remediation_detail and tag == 'div':
            if self.current_vuln:
                self.current_vuln['remediation'] = text
            self.in_remediation_detail = False
        elif self.in_remediation_background and tag == 'div':
            if self.current_vuln:
                self.current_vuln['remediation'] += ' ' + text
            self.in_remediation_background = False
        elif self.in_severity and tag in ['span', 'div', 'td']:
            if self.current_vuln:
                self.current_vuln['severity'] = text
            self.in_severity = False
        elif self.in_url and tag == 'a':
            if self.current_vuln:
                self.current_vuln['url'] = text
            self.in_url = False

    def handle_data(self, data):
        if any([self.in_issue_name, self.in_issue_detail, self.in_issue_background,
                self.in_remediation_detail, self.in_remediation_background,
                self.in_severity, self.in_url]):
            self.captured_text += data

    def finalize(self):
        """Call after feeding all data to capture the last vulnerability."""
        if self.current_vuln and self.current_vuln.get('name'):
            self.vulnerabilities.append(self.current_vuln)
        return self.vulnerabilities


def load_burp_data(burp_file_path):
    """Parse Burp HTML report and build lookup dictionary for enrichment."""
    print(f"\n=== Loading Burp Data ===")
    print(f"Parsing: {burp_file_path}")

    burp_data = {}

    try:
        with open(burp_file_path, 'r', encoding='utf-8', errors='replace') as f:
            html_content = f.read()

        parser = BurpHTMLParser()
        parser.feed(html_content)
        vulnerabilities = parser.finalize()

        # Also try regex-based fallback extraction for varied Burp formats
        if not vulnerabilities:
            print("  HTML parser found no structured data, trying regex fallback...")
            # Common Burp report patterns
            issue_blocks = re.findall(
                r'<h[123][^>]*>([^<]+)</h[123]>.*?(?:severity[^>]*>([^<]+)<|'
                r'class="[^"]*severity[^"]*"[^>]*>([^<]+)<).*?'
                r'(?:detail|description)[^>]*>(.*?)</div>.*?'
                r'(?:remediation|solution)[^>]*>(.*?)</div>',
                html_content, re.DOTALL | re.IGNORECASE
            )
            for block in issue_blocks:
                name = re.sub(r'<[^>]+>', '', block[0]).strip()
                severity = (block[1] or block[2]).strip()
                detail = re.sub(r'<[^>]+>', ' ', block[3]).strip()
                remediation = re.sub(r'<[^>]+>', ' ', block[4]).strip()
                if name:
                    vulnerabilities.append({
                        'name': name, 'severity': severity,
                        'detail': detail, 'background': '',
                        'remediation': remediation, 'url': ''
                    })

        findings_indexed = 0
        for vuln in vulnerabilities:
            name = vuln.get('name', '').strip()
            if not name:
                continue

            # Create multiple lookup keys for flexible matching
            name_key = name.lower().replace(' ', '_')
            burp_data[name_key] = vuln
            burp_data[name.lower()] = vuln

            # Also index by partial name for fuzzy matching
            words = name.lower().split()
            if len(words) >= 2:
                burp_data[' '.join(words[:3])] = vuln

            findings_indexed += 1

        print(f"Vulnerabilities parsed: {findings_indexed}")
        print(f"Lookup keys created: {len(burp_data)}")
        return burp_data

    except Exception as e:
        print(f"ERROR: Failed to load Burp file: {e}")
        return {}


def get_burp_enrichment(burp_data, data_field, source=''):
    """Look up Burp enrichment data using multiple fallback strategies."""
    if not burp_data:
        return None
    data_lower = str(data_field).lower().strip()

    # Strategy 1: Direct lookup by data field content
    for key in burp_data:
        if key in data_lower or data_lower in key:
            val = burp_data[key]
            if isinstance(val, dict):
                return val

    # Strategy 2: Match by vulnerability name fragments
    for key, vuln in burp_data.items():
        if isinstance(vuln, dict) and vuln.get('name', '').lower() in data_lower:
            return vuln

    # Strategy 3: Match by first significant words
    data_words = re.findall(r'\w+', data_lower)
    if len(data_words) >= 2:
        search_key = ' '.join(data_words[:3])
        if search_key in burp_data:
            val = burp_data[search_key]
            if isinstance(val, dict):
                return val

    return None


def format_burp_description(enrichment, source, data_field, severity):
    """Format Burp description for output. Ensures 3-5 sentences."""
    if not enrichment:
        return None
    detail = enrichment.get('detail', '') or enrichment.get('background', '')
    raw = ' '.join(detail.split()) if detail else ''

    desc = f"A {severity.lower()} severity web application vulnerability '{enrichment.get('name', data_field[:80])}' was identified on {source}. "
    if raw:
        if len(raw) > 800:
            raw = raw[:800]
            lp = raw.rfind('.')
            raw = raw[:lp+1] if lp > 400 else raw + "..."
        desc += raw + " "
    desc += "Web application vulnerabilities of this type can be exploited to compromise user sessions, exfiltrate sensitive data, or gain unauthorized access to application functionality. "
    url = enrichment.get('url', '')
    if url and url not in ['nan', '']:
        desc += f"Affected URL: {url}. "
    desc += "Immediate assessment and remediation is recommended based on the severity and exposure of the affected endpoint."
    return desc


def format_burp_recommendation(enrichment, source, severity):
    """Format Burp solution/recommendation. Ensures 3-5 sentences."""
    if not enrichment:
        return None
    raw = ' '.join(enrichment.get('remediation', '').split()) if enrichment.get('remediation') else ''
    if raw:
        if len(raw) > 600:
            raw = raw[:600]
            lp = raw.rfind('.')
            raw = raw[:lp+1] if lp > 300 else raw + "..."
        rec = raw + " "
    else:
        rec = f"Apply the vendor-recommended remediation for this vulnerability on {source}. "
    rec += "Verify remediation by re-scanning with web application security testing tools. "
    rec += f"Priority: {severity} - address according to organizational vulnerability management SLAs. "
    rec += "Consider implementing a Web Application Firewall (WAF) as a defense-in-depth measure while permanent fixes are applied. "
    rec += "Include this endpoint in regular dynamic application security testing (DAST) scans."
    return rec

print("\n=== CONSOLIDATION PHASE ===")

groups = defaultdict(list)
for f in all_findings:
    key = (f['Category'], f['Vuln_Type'], f['Priority'])
    groups[key].append(f)

print(f"Consolidation groups: {len(groups)}")
for key, findings in groups.items():
    if len(findings) > 1:
        print(f"  {key}: {len(findings)} findings")

consolidated_findings = []
os.makedirs('./output/findings_details', exist_ok=True)

for key, group_findings in groups.items():
    if len(group_findings) == 1:
        consolidated_findings.append(group_findings[0])
    else:
        category, vuln_type, priority = key
        assets = [f['Asset'] for f in group_findings]
        unique_assets = list(set([a for a in assets if a and a.lower() not in ['nan', 'none']]))
        asset_count = len(unique_assets)
        instance_count = len(group_findings)

        asset_list = ", ".join([f"({i+1}) {unique_assets[i]}" for i in range(min(5, len(unique_assets)))])
        if len(unique_assets) > 5:
            asset_list += f", and {len(unique_assets)-5} more"

        tracking_counts = defaultdict(int)
        for f in group_findings:
            tracking_counts[f.get('Tracking_Status', 'OPEN')] += 1
        tracking_summary = ", ".join([f"{k}: {v}" for k, v in tracking_counts.items()])

        avg_risk = None
        risk_scores = [f.get('Avg_Risk') for f in group_findings if f.get('Avg_Risk') is not None]
        if risk_scores:
            avg_risk = sum(risk_scores) / len(risk_scores)

        vuln_name = vuln_type.replace('_', ' ').title()
        _lead = f"{vuln_name} was identified on {instance_count} instances across {asset_count} assets: {asset_list}."
        _default_desc = (
            _lead + " This represents a systematic security issue affecting multiple systems "
            "requiring coordinated remediation. The widespread nature indicates common "
            "misconfiguration, shared vulnerable components, or consistent security control gaps."
        )
        _default_rec = (
            f"Implement systematic remediation across all {instance_count} affected instances. "
            "Deploy consistent security controls and monitoring across all affected assets. "
            f"Prioritize remediation based on asset criticality and verify each of the "
            f"{asset_count} affected assets through targeted testing."
        )
        description, recommendation = _enrich(vuln_type, _lead, _default_desc, _default_rec)

        consolidated = {
            'Category': category,
            'Tab': group_findings[0]['Tab'],
            'Priority': priority,
            'Item': f"{vuln_name} - {instance_count} instances across {asset_count} assets",
            'Description': description,
            'Recommendation': recommendation,
            'Vuln_Type': vuln_type,
            'Asset': ", ".join(unique_assets[:5]) if unique_assets else '',
            'Plugin_Name': group_findings[0].get('Plugin_Name', ''),
            'Tracking_Status': max(tracking_counts, key=tracking_counts.get),
        }
        if avg_risk is not None:
            consolidated['Avg_Risk'] = round(avg_risk, 1)
        consolidated_findings.append(consolidated)

        # Create detail file
        import csv as csv_mod
        detail_data = [{'Instance': i+1, 'Asset': f['Asset'],
                        'Details': f['Item'], 'Description': f['Description'][:500],
                        'Tracking_Status': f.get('Tracking_Status', 'OPEN'),
                        'FP_Status': f.get('FP_Status', 'Unverified')}
                       for i, f in enumerate(group_findings)]
        detail_file = f'./output/findings_details/findings_detail_{vuln_type}.csv'
        with open(detail_file, 'w', newline='') as csvf:
            writer = csv_mod.DictWriter(csvf, fieldnames=detail_data[0].keys())
            writer.writeheader()
            writer.writerows(detail_data)
        print(f"  Created: {detail_file}")

print(f"\nConsolidation complete: {len(all_findings)} -> {len(consolidated_findings)} findings")
